75 void FileGui::keyPressEvent(QKeyEvent* e)
76 {
77  if(c && c->popup()->isVisible()) {
78  // The following keys are forwarded by the completer to the widget
79  switch (e->key()) {
80  case Qt::Key_Enter:
81  case Qt::Key_Return:
82  case Qt::Key_Escape:
83  case Qt::Key_Tab:
84  case Qt::Key_Backtab:
85  e->ignore();
86  return; // let the completer do default behavior
87  default:
88  break;
89  }
90  }
91 
92  bool isShortcut = ((e->modifiers() & Qt::ControlModifier) && e->key() == Qt::Key_E); // CTRL+E
93  if (!c || !isShortcut) // dont process the shortcut when we have a completer
94  QPlainTextEdit::keyPressEvent(e);
95 
96  const bool ctrlOrShift = e->modifiers() & (Qt::ControlModifier | Qt::ShiftModifier);
97  if (!c || (ctrlOrShift && e->text().isEmpty()))
98  return;
99 
100  static QString eow("~!@#$%^&*()_+{}|:\"<>?,./;'[]\\-="); // end of word
101  bool hasModifier = (e->modifiers() != Qt::NoModifier) && !ctrlOrShift;
102  QString completionPrefix = textUnderCursor();
103 
104  if (!isShortcut && (hasModifier || e->text().isEmpty()|| completionPrefix.length() < 3
105  || eow.contains(e->text().right(1)))) {
106  c->popup()->hide();
107  return;
108  }
109 
110  if (completionPrefix != c->completionPrefix()) {
111  c->setCompletionPrefix(completionPrefix);
112  c->popup()->setCurrentIndex(c->completionModel()->index(0, 0));
113  }
114  QRect cr = cursorRect();
115  cr.setWidth(c->popup()->sizeHintForColumn(0)
116  + c->popup()->verticalScrollBar()->sizeHint().width());
117  c->complete(cr); // popup it up!
118 }
